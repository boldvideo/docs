## Client Types

### ClientOptions

```typescript
interface ClientOptions {
  baseURL?: string;
  debug?: boolean;

```

## Video Types

### Video

```typescript
interface Video {
  id: string;
  playbackId: string;
  title: string;
  description?: string;
  duration: number;
  status: string;

  createdAt: string;
  updatedAt: string;
  captions?: VideoSubtitles[];

  transcript?: VideoTranscript;

  metadata?: VideoMetadata;
}
```

### AIResponse

Non-streaming response from AI methods.

```typescript
interface AIResponse {
  text: string;
  sources?: Segment[];
  usage?: AIUsage;
}
```

### AIUsage

Token usage statistics.

```typescript
interface AIUsage {
  inputTokens: number;
  outputTokens: number;
}
```

### AIEvent

Union type for all streaming events.

```typescript
type AIEvent =
  | { type: 'message_start'; conversationId: string }
  | { type: 'text_delta'; text: string }
  | { type: 'sources'; sources: Segment[] }
  | { type: 'recommendations'; recommendations: Recommendation[] }
  | { type: 'message_complete'; usage: AIUsage }
  | { type: 'error'; error: string };
```

### Segment

Video segment used as a source/citation.

```typescript
interface Segment {

  videoTitle: string;
  start: number;
  end: number;
  text: string;
  speaker?: string;
}
```
